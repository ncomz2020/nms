$('#aletTest_confirm2').click(function () { swal({ title: "정말 삭제하시겠습니까?", type: "warning", showCancelButton: true, confirmButtonText: "예", cancelButtonText: "아니오", closeOnConfirm: false, closeOnCancel: false },function (isConfirm) { if (isConfirm) { swal({ title: "삭제 되었습니다", type: "success" confirmButtonText: "닫기" }); } else { swal({ title: "취소 되었습니다", type: "error" confirmButtonText: "닫기" }); } }); });
